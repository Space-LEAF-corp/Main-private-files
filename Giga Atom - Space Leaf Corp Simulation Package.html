Quick Start

Get up and running with Giga Atom in minutes

# Import the giga_atom package
from giga_atom import (
    ShellState, compute_excess_fraction, compute_lambda,
    baseline_counts, apply_overload, run_scenario,
    gyroscopic_S_time_series, bonding_scan
)

# Run a stability simulation
counts = baseline_counts()
overloaded = apply_overload(counts, {3: 0.5, 5: 0.3})
result = run_scenario("Stress Test", overloaded, days=3650)

# Analyze gyroscopic stability
t, S, I_total = gyroscopic_S_time_series(
    counts, days=3650, omega0=10.0, impulse_every=365
)

# Scan bonding compatibility for all elements
df = bonding_scan(val_giga=1, maxZ=118)
