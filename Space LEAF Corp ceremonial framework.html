3. Family Recognition Protocol (FRP)

  • Goal: Determine if a presence token belongs to a known family member without storing biometrics.
  • Scope: Only compares tokens within a single familyGroupId.
  • Reset: On ID replacement or explicit reset, all cached presence hints are invalidated.
// Pseudocode: FRP validation
function validateFamilyPresence(token: PresenceToken, familyMembers: FamilyMember[]): {
  recognized: boolean;
  role?: "steward" | "adult" | "child";
} {
  if (Date.now() > token.expiresAt) return { recognized: false };

  // Implementation detail: this could be a similarity check against
  // short-lived, per-session presence hints, not raw biometrics.
  const match = matchPresenceWithinFamily(token, familyMembers);

  if (!match) return { recognized: false };
  return { recognized: true, role: match.role };
}
