
terraform {
  required_version = ">= 1.9.3"
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "5.80.0"
    }
  }

}

variable "region" {
  description = "AWS region for this deployment"
  type        = string
  default     = "us-east-2"
}

provider "aws" {
  region = var.region
  default_tags {
    tags = {
      Application = "cloudflare-cds"

    }
  }
}


module "cloudflare_cds" {
  source = "git@github.com:cloudflare/cf-product-infrastructure-templates//cds/terraform/aws?ref=cds-latest"
}
